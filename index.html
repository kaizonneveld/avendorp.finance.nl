#!/usr/bin/env bash # deploy.sh — Wijnverdeling website naar GitHub Pages # Gebruik: bash deploy.sh [repo-naam] # Vereist: git, gh CLI (optioneel voor auto-repo aanmaken) set -e REPO_NAME="${1:-wijnverdeling}" FILE="wijnverdeling.html" BRANCH="gh-pages" echo "🍷 Wijnverdeling — GitHub Pages Deploy" echo "========================================" # 1. Controleer of het bestand bestaat if [ ! -f "$FILE" ]; then echo "❌ $FILE niet gevonden. Zet dit script naast het HTML-bestand." exit 1 fi # 2. Controleer git if ! command -v git &> /dev/null; then echo "❌ git is niet geïnstalleerd." exit 1 fi # 3. Init git repo als nog niet gedaan if [ ! -d ".git" ]; then echo "→ git init..." git init git branch -M main fi # 4. Kopieer HTML naar index.html (GitHub Pages serveert index.html als root) cp "$FILE" index.html echo "→ Gekopieerd naar index.html" # 5. Voeg remote toe (als nog niet bestaat) if ! git remote | grep -q origin; then echo "" echo "Voer je GitHub-gebruikersnaam in:" read -r GH_USER REMOTE_URL="https://github.com/${GH_USER}/${REPO_NAME}.git" # Probeer repo aan te maken via gh CLI if command -v gh &> /dev/null; then echo "→ Aanmaken GitHub repository '$REPO_NAME'..." gh repo create "$REPO_NAME" --public --confirm 2>/dev/null || echo " (repo bestaat al, doorgaan...)" else echo "" echo "⚠️ gh CLI niet gevonden. Maak handmatig een repository aan:" echo " https://github.com/new → naam: $REPO_NAME → Public" echo "" echo "Druk op Enter als de repo aangemaakt is..." read -r fi git remote add origin "$REMOTE_URL" echo "→ Remote ingesteld: $REMOTE_URL" fi # 6. Commit & push naar main git add index.html "$FILE" 2>/dev/null || git add index.html git commit -m "🍷 Wijnverdeling update $(date '+%Y-%m-%d %H:%M')" || echo " (niets te committen)" git push -u origin main 2>/dev/null || git push origin main # 7. Activeer GitHub Pages op main branch if command -v gh &> /dev/null; then echo "→ GitHub Pages activeren..." gh api repos/:owner/"$REPO_NAME"/pages \ --method POST \ --field source='{"branch":"main","path":"/"}' \ 2>/dev/null || \ gh api repos/:owner/"$REPO_NAME"/pages \ --method PUT \ --field source='{"branch":"main","path":"/"}' \ 2>/dev/null || \ echo " (Pages al actief of handmatig activeren in repo Settings > Pages)" fi echo "" echo "✅ Klaar!" echo "" REMOTE=$(git remote get-url origin) OWNER=$(echo "$REMOTE" | sed 's/.*github.com[:/]\(.*\)\/.*/\1/') echo "🌐 Website URL (even wachten op deploy ~1-2 min):" echo " https://${OWNER}.github.io/${REPO_NAME}/" echo "" echo "📋 Handmatig Pages activeren (als automatisch mislukt):" echo " Ga naar: $REMOTE → Settings → Pages" echo " Source: Deploy from branch → Branch: main → / (root)" echo "" echo "🔄 Toekomstige updates:" echo " bash deploy.sh"
